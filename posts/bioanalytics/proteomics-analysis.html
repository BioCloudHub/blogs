差异表达分析

library(DESeq2)

differential_analysis <- function(count_matrix, metadata) {
  dds <- DESeqDataSetFromMatrix(countData = count_matrix, colData = metadata, design = ~ condition)
  dds <- DESeq(dds)
  results <- results(dds, contrast = c("condition", "treatment", "control"))
  return(results)
}
